# Domain-Specific AI Agents

## Why This Matters

Generic AI agents are powerful, but domain-specific agents that deeply understand industry context, regulations, and workflows create 10x more value. Companies like Wisq (HR AI), Harvey (Legal AI), and Doximity (Healthcare AI) are building billion-dollar businesses by specializing agents for specific domains.

**What you'll learn:**
- How to build domain expertise into AI agents
- Industry-specific patterns (HR, healthcare, finance, legal)
- Compliance and safety guardrails by domain
- Real-world case studies and architectures

interface LegalAgent {
  domain: 'Legal',

  // Legal expertise
  expertise: {
    caselaw: 'Federal and state court decisions',
    statutes: 'U.S. Code, state statutes, regulations',
    contracts: 'Common contract types and clauses',
    legal_analysis: 'Legal reasoning and precedent analysis'
  },

  capabilities: [
    {
      name: 'legal_research',
      description: 'Find relevant caselaw, statutes, and precedents',
      implementation: async (legalIssue: string, jurisdiction: string) => {
        // Search legal databases (Westlaw, LexisNexis APIs)
        const cases = await legalDB.searchCases({
          query: legalIssue,
          jurisdiction: jurisdiction,
          relevance: 'high',
          recency: 'prefer_recent'
        })

        return {
          relevantCases: cases.map(c => ({
            citation: c.citation,
            summary: c.summary,
            relevance: c.relevanceScore,
            precedentialValue: c.bindingOrPersuasive
          })),
          statutes: await legalDB.searchStatutes(legalIssue, jurisdiction),
          analysis: await this.analyzeApplicability(cases, legalIssue)
        }
      }
    },
    {
      name: 'contract_review',
      description: 'Review contracts for risks and unusual clauses',
      implementation: async (contract: Document) => {
        const analysis = {
          unusualClauses: [] as ContractClause[],
          missingStandardClauses: [] as string[],
          riskAssessment: {
            liability: 'moderate',
            termination: 'favorable',
            indemnification: 'one-sided (unfavorable)'
          },
          recommendations: [] as string[]
        }

        // Extract and analyze clauses
        const clauses = await this.extractClauses(contract)

        // Flag unusual terms
        for (const clause of clauses) {
          const isStandard = await this.compareToMarket(clause)
          if (!isStandard) {
            analysis.unusualClauses.push({
              clause: clause.text,
              concern: clause.riskLevel,
              recommendation: clause.suggestedRevision
            })
          }
        }

        // Check for missing provisions
        const standardClauses = ['force_majeure', 'confidentiality', 'termination', 'governing_law']
        for (const required of standardClauses) {
          if (!clauses.find(c => c.type === required)) {
            analysis.missingStandardClauses.push(required)
          }
        }

        return analysis
      }
    },
    {
      name: 'generate_legal_memo',
      description: 'Draft legal memorandum on issue',
      implementation: async (facts: string, legalQuestion: string) => {
        // Research relevant law

          application: await this.applyLawToFacts(research, facts),
          conclusion: await this.synthesizeConclusion(research, facts),
          citations: research.relevantCases.map(c => c.citation)
        }
      }
    }

### Step 4: Implement Compliance Guardrails

```typescript
class DomainComplianceLayer {
  // Data access controls
  async enforceDataAccess(userId: string, dataType: string) {
    const permissions = await this.getPermissions(userId)

    // Domain-specific rules
    if (dataType === 'PHI' && !permissions.includes('HIPAA_authorized')) {
      throw new Error('HIPAA violation: User not authorized for PHI access')
    }

    if (dataType === 'MNPI' && !permissions.includes('insider_authorized')) {
      throw new Error('SEC violation: Unauthorized MNPI access')
    }

    // Audit log
    await this.auditLog.record({
      userId,
      dataType,
      timestamp: new Date(),
      action: 'access_granted'
    })
  }

  // Escalation triggers
  shouldEscalate(context: AgentContext): boolean {
    const triggers = {
      HR: ['discrimination', 'termination', 'legal_question'],
      Healthcare: ['diagnosis', 'prescription', 'life_threatening'],
      Legal: ['court_deadline', 'privileged_communication'],
      Finance: ['material_non_public', 'suspicious_activity']
    }

    const domainTriggers = triggers[context.domain]
    return domainTriggers.some(trigger =>
      context.conversation.includes(trigger)
    )
  }

  // Response validation
  async validateResponse(response: string, domain: string): Promise<void> {
    const violations = []

    // Check for prohibited phrases
    const prohibited = {
      Healthcare: ['I diagnose', 'You should take', 'Stop taking'],
      Legal: ['I guarantee', 'You will win', 'This is legal advice'],
      Finance: ['guaranteed return', 'risk-free', 'insider tip']
    }

    for (const phrase of prohibited[domain] || []) {
      if (response.toLowerCase().includes(phrase.toLowerCase())) {
        violations.push(`Prohibited phrase: "${phrase}"`)
      }
    }

    if (violations.length &gt; 0) {
      throw new Error(`Response validation failed: ${violations.join(', ')}`)
    }
  }
}
```
